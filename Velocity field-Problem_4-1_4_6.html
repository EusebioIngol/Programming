import numpy as np
import matplotlib.pyplot as plt
%matplotlib inline
x,y = np.meshgrid(np.linspace(-3,3,10),np.linspace(-1,5,10))
u = (0.5 + 0.8*x)
v = (1.5 -0.8*y)
plt.quiver(x,y,u,v)
plt.show()
